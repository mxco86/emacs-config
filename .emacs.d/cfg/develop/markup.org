      :commands (yaml-mode)
      :init
      (add-hook 'yaml-mode-hook 'flycheck-mode)
      (add-hook 'yaml-mode-hook 'flyspell-prog-mode))

    (use-package flycheck-yamllint
      :ensure t
      :after yaml-mode
      :commands (yaml-mode)
      :init
