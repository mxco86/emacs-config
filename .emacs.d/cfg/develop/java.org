* Java Development
  #+begin_src emacs-lisp
  (defun mjr/enable-eglot-java-hook ()
    (interactive)
    (add-hook 'java-mode-hook 'eglot-ensure))

  (defun mjr/disable-eglot-java-hook ()
    (interactive)
    (remove-hook 'java-mode-hook 'eglot-ensure))

  (use-package groovy-mode
    :ensure t
    :mode "\\.gradle\\'")
  #+end_src
